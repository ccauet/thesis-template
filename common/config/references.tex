% BibLaTeX
\usepackage[backend=biber,%
            citestyle=numeric-comp,%
            %bibstyle=draft,% numeric
            sorting=none,%
            firstinits=true,%
           ]{biblatex}
\addglobalbib{common/references/bibliography.bib}

% suppressing specific entry fields
\AtEveryBibitem{%
  \clearname{translator}%
  \clearname{comments}%
  \clearfield{month}%
  \clearfield{pagetotal}%
}
% suppressing "In:" 
\renewbibmacro{in:}{}
% et al. in italics
\DefineBibliographyStrings{english}{andothers=\textit{et~al\adddot}}
% title in italic
\DeclareFieldFormat[article,book,thesis,online,unpublished]{title}{\textit{#1}}
% volume in bold face
\DeclareFieldFormat[article,book,thesis,online,unpublished]{volume}{\textbf{#1}}
% journal in roman face
\DeclareFieldFormat[article,book,thesis,online,unpublished]{journaltitle}{#1}
% space before pages field
\renewcommand*{\bibpagespunct}{\addspace}
% no prefix for pages field
\DeclareFieldFormat[article,book,thesis,online,unpublished]{pages}{#1}
% comma as delimiter
\renewcommand{\newunitpunct}{\addcomma\space}

    linktoc=all,
    }
